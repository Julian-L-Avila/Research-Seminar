\begin{frame}{\textbf{Research Questions and Hypotheses}}

  \onslide<1->{\textbf{Purpose:} Define the \alert{main goal} of a study.}

  \onslide<2->{\emph{Driven by current trends and advances in the field.}}

  \onslide<3->{\textbf{Research Questions (RQs)} start as:
    \begin{itemize}
      \item<4-> Descriptive inquiries
      \item<5-> Later refined into \alert{inferential questions}
  \end{itemize}}

  \onslide<6->{They must be \textbf{specific} and \textbf{concise} to support hypothesis development.}

\end{frame}

\begin{frame}{\textbf{Hypotheses: Formal Predictions}}

  \onslide<1->{\textbf{Hypotheses} propose \alert{predicted outcomes}.}

  \onslide<2->{They define \emph{expected relationships} between variables or groups.}

  \onslide<3->{
    \begin{block}{Key Traits}
      \begin{itemize}
        \item Formal and testable
        \item Based on specific research questions
        \item \alert{Guide data collection and analysis}
      \end{itemize}
  \end{block}}

\end{frame}

\begin{frame}{\textbf{Quantitative vs Qualitative RQs and Hypotheses}}

  \onslide<1->{\textbf{Quantitative Research:}
    \begin{itemize}
      \item<2-> RQs framed at the start
      \item<3-> Inquire about \alert{variable relationships}
      \item<4-> Hypotheses predict:
        \begin{itemize}
          \item<5-> Simple (1 DV \& 1 IV)
          \item<6-> Complex (multiple DVs/IVs)
        \end{itemize}
  \end{itemize}}

  \vspace{0.5cm}

  \onslide<7->{\textbf{Qualitative Research:}
    \begin{itemize}
      \item<8-> RQs evolve during the study
      \item<9-> Emphasis on \emph{central and sub-questions}
      \item<10-> Hypotheses may:
        \begin{itemize}
          \item<11-> Be stated as \alert{problem statements}
          \item<12-> Lead to both testing and generating new hypotheses
        \end{itemize}
  \end{itemize}}

\end{frame}
