\begin{frame}{\textbf{Frameworks for Developing RQs and Hypotheses}}

  \onslide<1->{\textbf{Plan ahead:} RQs and hypotheses must be set \alert{before} the
  study begins.}

  \vspace{0.3cm}

  \onslide<2->{\textbf{Use the \alert{FINER} criteria:}}

  \begin{itemize}
    \item<3-> \textbf{F}easible
    \item<4-> \textbf{I}nteresting
    \item<5-> \textbf{N}ovel
    \item<6-> \textbf{E}thical
    \item<7-> \textbf{R}elevant
  \end{itemize}

  \vspace{0.3cm}

  \onslide<8->{\emph{A structured foundation leads to focused and meaningful research.}}

\end{frame}

\begin{frame}{\textbf{Constructing RQs and Hypotheses (1)}}

  \onslide<1->{\textbf{1. Clarify} the background of the study.}

  \onslide<2->{\textbf{2. Identify} the research problem \emph{early and clearly}.}

  \onslide<3->{\textbf{3. Review} existing literature and conduct preliminary research:}

  \begin{itemize}
    \item<4-> Explore related \emph{theories}
    \item<5-> Analyze relevant \emph{previous studies}
  \end{itemize}

  \vspace{0.3cm}

  \onslide<6->{These steps build the \alert{knowledge base} for formulating your RQs.}

\end{frame}

\begin{frame}{\textbf{Constructing RQs and Hypotheses (2)}}

  \onslide<1->{\textbf{4. Formulate} research questions to investigate the problem.}

  \onslide<2->{\textbf{5. Identify} variables and define them \emph{operationally}.}

  \begin{itemize}
    \item<3-> Variables should be accessible and measurable
    \item<4-> Definitions must connect to the research context
  \end{itemize}

  \vspace{0.3cm}

  \onslide<5->{\textbf{6. Develop} hypotheses from your RQs:}

  \begin{itemize}
    \item<6-> Deductive: from general theory to specific prediction
    \item<7-> Inductive: from data patterns to hypothesis formation
  \end{itemize}

  \onslide<8->{\textbf{7. State} the study aims clearly.}

\end{frame}

\begin{frame}{\textbf{Summary: RQs First, Hypotheses Next}}

  \onslide<1->{\textbf{Research questions} are built from the study background.}

  \onslide<2->{\textbf{Hypotheses} follow and are based directly on those questions.}

  \vspace{0.3cm}

  \onslide<3->{
  \begin{block}{Why This Matters}
    \begin{itemize}
      \item<4-> Strong RQs lead to \alert{strong hypotheses}
      \item<5-> Clear hypotheses define \textbf{objectives}, \textbf{design},
                and \textbf{outcomes}
    \end{itemize}
  \end{block}}

  \vspace{0.2cm}

  \onslide<6->{\emph{Good questions drive good science.}}

\end{frame}
